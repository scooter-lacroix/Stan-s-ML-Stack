- [ ] **Step 1: Write the helper library**

```bash
#!/bin/bash
# ui_installer_helper.sh - Shared functions for UI component installers
# Used by: install_comfyui.sh, install_vllm_studio.sh, install_textgen.sh
#
# Provides:
#   ui_parse_common_args DIR_VAR REPO_URL_VAR DRY_RUN_VAR
#   ui_git_clone_or_update INSTALL_DIR REPO_URL [PRESERVE_DIRS...]
#   ui_fix_ownership INSTALL_DIR
#   ui_create_launcher_shim SHIM_NAME LAUNCHER_DIR LAUNCH_COMMAND DESCRIPTION
#   ui_create_systemd_service SERVICE_NAME INSTALL_DIR EXEC_COMMAND ENV_VARS
#   ui_detect_gpu_devices
#   ui_print_summary COMPONENT_NAME INSTALL_DIR COMMANDS_ARRAY

# Parse common CLI arguments: --dry-run, --dir <path>, --help
# Usage: ui_parse_common_args DIR_VAR REPO_URL_VAR DRY_RUN_VAR
# Sets the caller's variables by name using nameref (bash 4.3+)
ui_parse_common_args() {
    local -n _dir_var="$1"
    local -n _repo_var="$2"
    local -n _dry_var="$3"

    while [[ $# -gt 3 ]]; do
        case "$4" in
            --dry-run)
                _dry_var=true
                shift
                ;;
            --dir)
                _dir_var="$5"
                shift 2
                ;;
            --help|-h)
                echo "Usage: $0 [--dry-run] [--dir <path>]"
                exit 0
                ;;
            *)
                shift
                ;;
        esac
    done
}

# Clone a git repo or update an existing installation
# Usage: ui_git_clone_or_update INSTALL_DIR REPO_URL [PRESERVE_DIRS...]
# Preserves listed directories during git reset --hard (e.g., models/, input/)
ui_git_clone_or_update() {
    local install_dir="$1"
    local repo_url="$2"
    shift 2
    local -a preserve_dirs=("$@")

    local has_content=false

    check_dir_has_content() {
        [ -d "$1" ] && [ "$(ls -A "$1" 2>/dev/null)" ]
    }

    if [ -d "$install_dir/.git" ]; then
        # Check for existing content in preserve dirs
        for dir in "${preserve_dirs[@]}"; do
            local full_path="$install_dir/$dir"
            if check_dir_has_content "$full_path"; then
                has_content=true
                break
            fi
        done

        if [ "$has_content" = true ]; then
            print_warning "Existing user data detected - will preserve during update"
        fi

        print_step "Updating $(basename "$install_dir")..."
        git -C "$install_dir" remote set-head origin -a || true
        local default_branch
        default_branch=$(git -C "$install_dir" symbolic-ref refs/remotes/origin/HEAD 2>/dev/null | sed 's@^refs/remotes/origin/@@')
        default_branch=${default_branch:-master}

        execute_command "git -C \"$install_dir\" checkout \"$default_branch\"" "Checking out $default_branch branch"
        execute_command "git -C \"$install_dir\" fetch --all" "Fetching updates"

        if [ "$has_content" = true ]; then
            # Build stash paths for preservation
            local -a stash_paths=()
            for dir in "${preserve_dirs[@]}"; do
                stash_paths+=("$install_dir/$dir")
            done

            execute_command "git -C \"$install_dir\" stash push -u -m \"rusty-stack-preserve-user-data\" -- ${stash_paths[*]}" "Stashing user data"
            execute_command "git -C \"$install_dir\" reset --hard \"origin/$default_branch\"" "Resetting to origin/$default_branch"
            execute_command "git -C \"$install_dir\" stash pop" "Restoring user data"
            print_success "User data preserved during update"
        else
            execute_command "git -C \"$install_dir\" reset --hard \"origin/$default_branch\"" "Resetting to origin/$default_branch"
        fi
    else
        execute_command "git clone \"$repo_url\" \"$install_dir\"" "Cloning $(basename "$install_dir") repository"
    fi
}

# Fix ownership when run with sudo
# Usage: ui_fix_ownership INSTALL_DIR
ui_fix_ownership() {
    local install_dir="$1"
    if [ "$EUID" -eq 0 ] && [ -n "${SUDO_USER:-}" ]; then
        chown -R "$SUDO_USER:$SUDO_USER" "$install_dir"
    fi
}

# Create a launcher shim script in ~/.local/bin/
# Usage: ui_create_launcher_shim SHIM_NAME LAUNCHER_DIR LAUNCH_COMMAND DESCRIPTION
# Example: ui_create_launcher_shim "comfy" "$HOME/.local/bin" "cd $COMFYUI_DIR && python3 main.py" "ComfyUI launcher"
ui_create_launcher_shim() {
    local shim_name="$1"
    local launcher_dir="$2"
    local launch_command="$3"
    local description="$4"
    local dry_run="${DRY_RUN:-false}"

    mkdir -p "$launcher_dir" 2>/dev/null || true

    if [ "$dry_run" = "false" ]; then
        local launcher_path="$launcher_dir/$shim_name"
        cat > "$launcher_path" << EOF
#!/bin/bash
# $description - Generated by Stan's ML Stack
$launch_command "\$@"
EOF
        chmod +x "$launcher_path"
        print_success "Launcher script created at $launcher_path"

        # Ensure ~/.local/bin is in PATH
        if [[ ":$PATH:" != *":$HOME/.local/bin:"* ]]; then
            print_warning "~/.local/bin is not in your PATH. Add it to your shell profile:"
            echo "  export PATH=\"\$HOME/.local/bin:\$PATH\""
        fi
    else
        print_step "[DRY-RUN] Would create launcher at $launcher_dir/$shim_name"
    fi
}

# Create a systemd user service file
# Usage: ui_create_systemd_service SERVICE_NAME INSTALL_DIR EXEC_COMMAND [ENV_VAR...]
# Example: ui_create_systemd_service "comfyui" "$COMFYUI_DIR" "$PYTHON_BIN main.py --enable-manager" "HIP_VISIBLE_DEVICES=$GPU_DEVICES"
ui_create_systemd_service() {
    local service_name="$1"
    local install_dir="$2"
    local exec_command="$3"
    shift 3
    local -a env_vars=("$@")
    local dry_run="${DRY_RUN:-false}"
    local service_dir="$HOME/.config/systemd/user"

    if [ "$dry_run" = "false" ] && [ -d "$service_dir" ]; then
        local env_lines=""
        for ev in "${env_vars[@]}"; do
            env_lines+="Environment=\"$ev\""$'\n        '
        done

        cat > "$service_dir/${service_name}.service" << EOF
[Unit]
Description=$service_name
After=network.target

[Service]
Type=simple
WorkingDirectory=$install_dir
${env_lines}ExecStart=$exec_command
Restart=on-failure

[Install]
WantedBy=default.target
EOF
        print_step "To enable as a service: systemctl --user enable --now ${service_name}.service"
    fi
}

# Detect AMD GPU devices for HIP_VISIBLE_DEVICES
# Usage: ui_detect_gpu_devices
# Outputs: comma-separated GPU device list (e.g., "0,1")
ui_detect_gpu_devices() {
    local gpu_devices=""

    if command -v rocm-smi &>/dev/null; then
        local gpu_count
        gpu_count=$(rocm-smi --showproductname 2>/dev/null | grep -c "GPU\[" || echo "0")
        if [ "$gpu_count" -gt 0 ]; then
            gpu_devices=$(seq -s, 0 $((gpu_count - 1)))
            print_step "Detected $gpu_count AMD GPU(s)"
        fi
    fi

    if [ -z "$gpu_devices" ]; then
        gpu_devices="0,1"
        print_warning "Could not detect GPU count, using default: $gpu_devices"
    fi

    echo "$gpu_devices"
}

# Print a formatted installation summary
# Usage: ui_print_summary COMPONENT_NAME INSTALL_DIR
# Reads COMMANDS_ARRAY for run commands to display
ui_print_summary() {
    local component_name="$1"
    local install_dir="$2"

    print_section "$component_name Installation Summary"
    print_success "$component_name installed at: $install_dir"
    echo
    echo -e "${BOLD}Run Commands:${RESET}"

    if declare -p COMMANDS_ARRAY &>/dev/null; then
        for cmd in "${COMMANDS_ARRAY[@]}"; do
            echo "  $cmd"
        done
    fi

    echo
    echo -e "${BOLD}GPU Acceleration:${RESET}"
    print_success "Uses ROCm for AMD GPU acceleration"
}
```

```bash
# update_helper.sh - Detection and update functions for rusty-stack update

SCRIPT_LIB_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

# Check if a Python module is importable
up_is_python_module() {
    local module="$1"
    local python="${2:-python3}"
    "$python" -c "import $module" &>/dev/null
}

# Check if a command exists
up_command_exists() {
    command -v "$1" >/dev/null 2>&1
}

# Check if a path exists
up_path_exists() {
    [ -e "$1" ]
}

# Get the user's home directory
up_user_home() {
    echo "${MLSTACK_USER_HOME:-${SUDO_USER:-${USER:-$(id -un)}}}"
    eval echo "~$(id -un)"
}

# Detect all installed components
# Returns a newline-separated list of component IDs
up_detect_installed() {
    local home
    home=$(up_user_home)
    local installed=()

    # ROCm
    if [ -f "/opt/rocm/.info/version" ] && up_command_exists rocminfo; then
        installed+=("rocm")
    fi

    # Python-based components
    for entry in \
        "pytorch:torch" \
        "triton:triton" \
        "mpi4py:mpi4py" \
        "deepspeed:deepspeed" \
        "vllm:vllm" \
        "aiter:aiter" \
        "onnx:onnxruntime" \
        "bitsandbytes:bitsandbytes" \
        "wandb:wandb"; do
        local comp_id="${entry%%:*}"
        local module="${entry##*:}"
        if up_is_python_module "$module"; then
            installed+=("$comp_id")
        fi
    done

    # Git-based components
    for entry in \
        "comfyui:$home/ComfyUI" \
        "vllm-studio:$home/vllm-studio" \
        "textgen:$home/text-generation-webui"; do
        local comp_id="${entry%%:*}"
        local dir="${entry##*:}"
        if [ -d "$dir/.git" ]; then
            installed+=("$comp_id")
        fi
    done

    # Flash Attention (special detection)
    if up_is_python_module "flash_attn" || up_is_python_module "flash_attn_2"; then
        installed+=("flash-attn")
    fi

    # ROCm SMI
    if up_command_exists "rocm-smi"; then
        installed+=("rocm-smi")
    fi

    # MIGraphX
    if up_is_python_module "migraphx"; then
        installed+=("migraphx")
    fi

    # Permanent env
    if [ -f "$home/.mlstack_env" ]; then
        installed+=("permanent-env")
    fi

    printf '%s\n' "${installed[@]}"
}
```

```bash
            echo "  rusty-stack update --list       # Show what's installed"
            exit 0
            ;;
        *)
            MODE="specific"
            SPECIFIC_COMPONENTS+=("$1")
            shift
            ;;
    esac
done

# --- List Mode ---
if [ "$MODE" = "list" ]; then
    echo -e "${CYAN}${BOLD}Installed Components:${RESET}"
    echo

    local installed_list
    installed_list=$(up_detect_installed)

    if [ -z "$installed_list" ]; then
        echo -e "${YELLOW}No components detected as installed.${RESET}"
        exit 0
    fi

    printf "  %-25s %-20s %s\n" "COMPONENT" "VERSION" "STATUS"
    printf "  %-25s %-20s %s\n" "-------------------------" "--------------------" "------"

    while IFS= read -r comp_id; do
        local name version
        name=$(up_display_name "$comp_id")
        version=$(up_get_version "$comp_id")
        printf "  ${GREEN}%-25s${RESET} %-20s ${GREEN}installed${RESET}\n" "$name" "$version"
    done <<< "$installed_list"

    echo
    exit 0
fi

# --- Detect installed components ---
installed_list=$(up_detect_installed)

if [ -z "$installed_list" ]; then
    echo -e "${RED}No components detected as installed.${RESET}"
    echo -e "${YELLOW}Run the Rusty Stack installer first.${RESET}"
    exit 1
fi

# Convert to array
mapfile -t INSTALLED_COMPONENTS <<< "$installed_list"

# --- All Mode ---
if [ "$MODE" = "all" ]; then
    echo -e "${CYAN}${BOLD}Updating all installed components...${RESET}"
    echo

    local failed=()
    local succeeded=()

    for comp_id in "${INSTALLED_COMPONENTS[@]}"; do
        local name
        name=$(up_display_name "$comp_id")
        echo -e "${BLUE}Updating $name...${RESET}"

        if up_update_component "$comp_id" "$SCRIPT_DIR" "--force"; then
            echo -e "${GREEN}  $name updated successfully${RESET}"
            succeeded+=("$comp_id")
        else
            echo -e "${RED}  $name update failed${RESET}"
            failed+=("$comp_id")
        fi
        echo
    done

    # Summary
    echo -e "${CYAN}${BOLD}Update Summary:${RESET}"
    echo -e "  ${GREEN}Succeeded: ${#succeeded[@]}${RESET}"
    for c in "${succeeded[@]}"; do
        echo -e "    ${GREEN}$(up_display_name "$c")${RESET}"
    done

    if [ ${#failed[@]} -gt 0 ]; then
        echo -e "  ${RED}Failed: ${#failed[@]}${RESET}"
        for c in "${failed[@]}"; do
            echo -e "    ${RED}$(up_display_name "$c")${RESET}"
        done
        exit 1
    fi

    exit 0
fi

# --- Specific Components Mode ---
if [ "$MODE" = "specific" ]; then
    echo -e "${CYAN}${BOLD}Updating specific components...${RESET}"
    echo

    for comp_id in "${SPECIFIC_COMPONENTS[@]}"; do
        local name
        name=$(up_display_name "$comp_id")

        # Check if installed
        local found=false
        for installed in "${INSTALLED_COMPONENTS[@]}"; do
            if [ "$installed" = "$comp_id" ]; then
                found=true
                break
            fi
        done

        if [ "$found" = false ]; then
            echo -e "${YELLOW}  $name is not installed, skipping${RESET}"
            continue
        fi

        echo -e "${BLUE}Updating $name...${RESET}"
        if up_update_component "$comp_id" "$SCRIPT_DIR" "--force"; then
            echo -e "${GREEN}  $name updated successfully${RESET}"
        else
            echo -e "${RED}  $name update failed${RESET}"
        fi
        echo
    done

    exit 0
fi

# --- Interactive Menu Mode ---
echo -e "${CYAN}${BOLD}Rusty Stack - Component Update${RESET}"
echo
echo -e "${BOLD}Installed components detected:${RESET}"
echo

# Display menu
for i in "${!INSTALLED_COMPONENTS[@]}"; do
    local comp_id="${INSTALLED_COMPONENTS[$i]}"
    local name version
    name=$(up_display_name "$comp_id")
    version=$(up_get_version "$comp_id")
    printf "  ${BOLD}%d)${RESET} %-25s ${YELLOW}[%s]${RESET}\n" "$((i + 1))" "$name" "$version"
done

echo
echo -e "  ${BOLD}a)${RESET} ALL - Update all installed components"
echo -e "  ${BOLD}q)${RESET} Quit"
echo
echo -e "Enter component numbers to update (comma-separated, e.g. 1,3,5):"
read -r -p "> " SELECTION

# Parse selection
if [[ "$SELECTION" =~ ^[Qq]$ ]]; then
    echo "Cancelled."
    exit 0
fi

SELECTED_COMPONENTS=()

if [[ "$SELECTION" =~ ^[Aa]$ ]]; then
    # ALL selected
    SELECTED_COMPONENTS=("${INSTALLED_COMPONENTS[@]}")
else
    # Parse comma/space-separated numbers
    IFS=', ' read -ra NUMBERS <<< "$SELECTION"
    for num in "${NUMBERS[@]}"; do
        # Validate number
        if [[ "$num" =~ ^[0-9]+$ ]] && [ "$num" -ge 1 ] && [ "$num" -le "${#INSTALLED_COMPONENTS[@]}" ]; then
            local idx=$((num - 1))
            SELECTED_COMPONENTS+=("${INSTALLED_COMPONENTS[$idx]}")
        else
            echo -e "${YELLOW}Invalid selection: $num (skipping)${RESET}"
        fi
    done
fi

if [ ${#SELECTED_COMPONENTS[@]} -eq 0 ]; then
    echo "No valid components selected."
    exit 0
fi

# Confirm
echo
echo -e "${BOLD}Components to update:${RESET}"
for comp_id in "${SELECTED_COMPONENTS[@]}"; do
    echo -e "  - $(up_display_name "$comp_id")"
done
echo
read -r -p "Proceed? (y/N): " CONFIRM
CONFIRM=$(echo "${CONFIRM:-n}" | tr '[:upper:]' '[:lower:]')
if [[ "$CONFIRM" != "y" && "$CONFIRM" != "yes" ]]; then
    echo "Cancelled."
    exit 0
fi

# Execute updates
echo
local failed=()
local succeeded=()

for comp_id in "${SELECTED_COMPONENTS[@]}"; do
    local name
    name=$(up_display_name "$comp_id")
    echo -e "${BLUE}Updating $name...${RESET}"

    if up_update_component "$comp_id" "$SCRIPT_DIR" "--force"; then
        echo -e "${GREEN}  $name updated successfully${RESET}"
        succeeded+=("$comp_id")
    else
        echo -e "${RED}  $name update failed${RESET}"
        failed+=("$comp_id")
    fi
    echo
done

# Summary
echo -e "${CYAN}${BOLD}Update Summary:${RESET}"
echo -e "  ${GREEN}Succeeded: ${#succeeded[@]}${RESET}"
echo -e "  ${RED}Failed: ${#failed[@]}${RESET}"

if [ ${#failed[@]} -gt 0 ]; then
    exit 1
fi

exit 0
```

```rust
//! rusty-stack-update - Component update CLI command
//!
//! Thin wrapper that detects the scripts directory and delegates to
//! scripts/update_stack.sh for the actual update logic.

use std::env;
use std::path::PathBuf;
use std::process::Command;

fn main() {
    let args: Vec<String> = env::args().skip(1).collect();

    let scripts_dir = detect_scripts_dir();

    let mut cmd = Command::new("bash");
    cmd.arg(format!("{}/update_stack.sh", scripts_dir));

    // Forward all arguments to the shell script
    for arg in &args {
        cmd.arg(arg);
    }

    // Inherit stdin/stdout/stderr for interactive menu
    cmd.stdin(std::process::Stdio::inherit())
        .stdout(std::process::Stdio::inherit())
        .stderr(std::process::Stdio::inherit());

    let status = cmd.status().unwrap_or_else(|e| {
        eprintln!("Failed to execute update script: {}", e);
        std::process::exit(1);
    });

    std::process::exit(status.code().unwrap_or(1));
}

fn detect_scripts_dir() -> String {
    let cwd = env::current_dir().unwrap_or_else(|_| PathBuf::from("."));
    let scripts = cwd.join("scripts");
    if scripts.exists() {
        return scripts.to_string_lossy().to_string();
    }
    let parent_scripts = cwd.join("..").join("scripts");
    if parent_scripts.exists() {
        return parent_scripts.to_string_lossy().to_string();
    }

    // Try relative to the binary location
    if let Ok(exe) = env::current_exe() {
        if let Some(exe_dir) = exe.parent() {
            // Binary is in rusty-stack/target/release/
            // Scripts are in ../../scripts/ (from repo root)
            let repo_root = exe_dir
                .parent()  // target/
                .and_then(|p| p.parent())  // rusty-stack/
                .and_then(|p| p.parent()); // repo root
            if let Some(root) = repo_root {
                let s = root.join("scripts");
                if s.exists() {
                    return s.to_string_lossy().to_string();
                }
            }
        }
    }

    "./scripts".to_string()
}
```
